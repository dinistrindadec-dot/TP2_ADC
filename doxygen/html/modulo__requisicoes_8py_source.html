62def texto_tempo_restante_devolucao(data_limite_iso):
63
64 if not data_limite_iso:
65 return "Prazo não definido nesta requisição (registo antigo)."
66 try:
67 limite = date.fromisoformat(data_limite_iso)
68 except ValueError:
69 return "Data limite inválida no registo."
70 hoje = date.today()
71 delta = (limite - hoje).days
72 limite_pt = limite.strftime("%d/%m/%Y")
73 if delta > 1:
74 return f"Faltam {delta} dias até ao limite de devolução ({limite_pt})."
75 if delta == 1:
76 return f"Falta 1 dia até ao limite de devolução ({limite_pt})."
77 if delta == 0:
78 return f"Hoje é o último dia do prazo ({limite_pt})."
79 return f"Em atraso há {abs(delta)} dia(s). Limite era {limite_pt}."

119def exemplares_disponiveis(livro_id):
120 """
121 Calcula quantos exemplares ainda podem ser requisitados.

124 :return: Número de exemplares livres ou 0 se o livro não existir
125 :rtype: int

134def requisitar(livro_id, utilizador_id):
135 """
136 Cria uma requisição ativa se houver exemplar disponível.
137
138 :param livro_id: Identificador do livro
139 :param utilizador_id: Identificador do utilizador registado
140 :return: Tupla (sucesso: bool, mensagem: str, requisicao ou None)
141 :rtype: tuple[bool, str, dict | None]
142 """
143 import modulo_utilizadores
144
145 utilizador = modulo_utilizadores.obter_utilizador(utilizador_id)
146 if utilizador is None:
147 return False, "Utilizador não encontrado. Peça ao administrador o registo.", None
148
149 livro = modulo_catalogo.obter_livro(livro_id)
150 if livro is None:
151 return False, "Livro não encontrado.", None
152 if exemplares_disponiveis(livro_id) <= 0:
153 return False, "Não há exemplares disponíveis para requisição.", None
154 regs = carregar_requisicoes()
155 hoje = date.today()
156 limite = hoje + timedelta(days=DIAS_PRAZO_EMPRESTIMO)
157 reg = {
158 "id": _proximo_id(regs),
159 "livro_id": livro_id,
160 "utilizador_id": utilizador_id,
161 "requisitante": utilizador["nome"],
162 "ativa": True,
163 "data_requisicao": hoje.isoformat(),
164 "data_limite": limite.isoformat(),
165 }
166 regs.append(reg)
167 _guardar_requisicoes(regs)
168 msg = (
169 "Requisição registada com sucesso. "
170 f"Devolução até {limite.strftime('%d/%m/%Y')} "
171 f"({DIAS_PRAZO_EMPRESTIMO} dias)."
172 )
173 return True, msg, reg

176def devolver(requisicao_id):
177 """
178 Encerra uma requisição (devolução de exemplar).
179
180 :param requisicao_id: Identificador da requisição
181 :return: Tupla (sucesso, mensagem)
182 :rtype: tuple[bool, str]
183 """
184 regs = carregar_requisicoes()
185 for r in regs:
186 if r["id"] == requisicao_id:
187 if not r["ativa"]:
188 return False, "Esta requisição já estava concluída."
189 r["ativa"] = False
190 _guardar_requisicoes(regs)
191 return True, "Devolução registada."
192 return False, "Requisição não encontrada."
